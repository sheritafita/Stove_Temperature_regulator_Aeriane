#include <multi_channel_relay.h>



Multi_Channel_Relay relay1;

void setup() {
Serial.begin(115200);
relay1.begin(0x12);
relay1.channelCtrl(0); // Turn off all channel


 /* Begin Controlling Relay */
    Serial.println("Channel 1 on");
    relay1.turn_on_channel(1);
    delay(500);
    Serial.println("Channel 2 on");
    relay1.turn_off_channel(1);
    relay1.turn_on_channel(2);
    delay(500);
    Serial.println("Channel 3 on");
    relay1.turn_off_channel(2);
    relay1.turn_on_channel(3);
    delay(500);
    Serial.println("Channel 4 on");
    Serial.println(relay1.getChannelState(), BIN);
    relay1.turn_off_channel(3);
    relay1.turn_on_channel(4);
    delay(500);
    relay1.turn_off_channel(4);
}
